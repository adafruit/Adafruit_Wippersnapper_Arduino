134  pos_start = 0; // start at the leftmost position of the display
135  } else {

140  if (message[i] != '.') {
141  seg_chars++;
142  }
143  }
144  // start at the rightmost position of the display
145  switch (seg_chars) {

147  pos_start = 0;

161  break;

167  for (size_t i = 0; i < len_display; i++) {
168 
169  // skip position 2

172  }
173  // Save the character because if there's a decimal, we need to skip it in
