119  // Calculate the number of characters to display
120  size_t len_display = min(strlen(message), (size_t)LED_MAX_CHARS);
121 
122  // Set the starting position based on alignment
123  int pos_start;
124  if (_alignment == LED_BACKPACK_ALIGNMENT_LEFT) {
125  pos_start = 0; // start at the leftmost position of the display
126  } else {
127  // Exclude decimal points from the character count because those get
128  // displayed on a "special" segment of the LED display
129  int seg_chars = 0;
130  for (size_t i = 0; i < len_display; i++) {
131  if (message[i] != '.') {
132  seg_chars++;
133  }
134  }
135  // start at the rightmost position of the display
136  pos_start = LED_MAX_CHARS - seg_chars;
137  }
138 
139  // Write to the display's buffer
140  int cur_idx = pos_start;
141  for (size_t i = 0; i < len_display; i++) {
142  // Save the character because if there's a decimal, we need to skip it in
143  // the buffer
144  char ch = message[i];
145 
146  // Look-ahead for a decimal point to attach to the current character
147  bool display_dot = false;
148  if (i + 1 < len_display && message[i + 1] == '.') {
149  display_dot = true;
150  i++;
151  len_display++;
152  } else if (message[i] == 0xC2 && message[i + 1] == 0xB0 &&
153  i + 1 < strlen(message)) {
154  // Write the degree symbol
155  _alpha4->writeDigitRaw(cur_idx, ALPHANUM_DEGREE_SYMBOL);
156  i++;
157  cur_idx++;
158  continue;
159  }
160  // Write the character to the display buffer
161  _alpha4->writeDigitAscii(cur_idx, ch, display_dot);
162  cur_idx++;
163  }
