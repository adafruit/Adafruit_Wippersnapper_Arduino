295  void _connect() {
296 
297  if (WiFi.status() == WL_CONNECTED)
298  return;
299 
300  if (strlen(_ssid) == 0) {
301  _status = WS_SSID_INVALID;
302  } else {
303  WiFi.setAutoReconnect(false);
304  _disconnect();
305  delay(100);
306  if (WS._isWiFiMulti) {
307  // multi network mode
308  _wifiMulti.APlistClean();
309  _wifiMulti.setAllowOpenAP(false);
310  // add default network
311  _wifiMulti.addAP(_ssid, _pass);
312  // add array of alternative networks
313  for (int i = 0; i < WS_MAX_ALT_WIFI_NETWORKS; i++) {
314  if (strlen(WS._multiNetworks[i].ssid) > 0) {
315  _wifiMulti.addAP(WS._multiNetworks[i].ssid,
316  WS._multiNetworks[i].pass);
317  }
318  }
319  if (_wifiMulti.run(20000) == WL_CONNECTED) {
320  _status = WS_NET_CONNECTED;
321  } else {
322  _status = WS_NET_DISCONNECTED;
323  }
324  } else {
325  // single network mode
326  WiFi.begin(_ssid, _pass);
327  _status = WS_NET_DISCONNECTED;
328  WS.feedWDT();
329  delay(5000);
330  }
331  WS.feedWDT();
332  }
333  }
