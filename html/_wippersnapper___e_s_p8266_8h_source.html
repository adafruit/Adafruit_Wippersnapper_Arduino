269  if (strlen(WS._multiNetworks[i].ssid) > 0 &&
270  (_wifiMulti.existsAP(WS._multiNetworks[i].ssid) == false)) {
271  // doesn't exist, add it
272  _wifiMulti.addAP(WS._multiNetworks[i].ssid,
